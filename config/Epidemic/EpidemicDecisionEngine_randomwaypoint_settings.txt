# Konfigurasi untuk Simulasi Epidemic Routing

## Pengaturan Skenario
Scenario.name = EpidemicSimulation_RandomWaypoint
# Aktifkan simulasi koneksi (penting untuk DTN)
Scenario.simulateConnections = true
# Interval pembaruan simulasi (semakin kecil, semakin akurat, tetapi semakin lambat)
Scenario.updateInterval = 0.1
# Waktu simulasi (detik) - 1 jam
Scenario.endTime = 43200

## Pengaturan Interface (Bluetooth)
btInterface.type = SimpleBroadcastInterface
# Kecepatan transmisi interface (bytes per detik) - 250kBps (cukup realistis untuk Bluetooth)
btInterface.transmitSpeed = 250k
# Jangkauan transmisi interface (meter) - 10 meter (sesuaikan dengan skenario Anda)
btInterface.transmitRange = 10

## Pengaturan Grup Host
# Jumlah grup host yang berbeda (dalam kasus ini, hanya satu grup)
Scenario.nrofHostGroups = 1

# Pengaturan umum untuk grup host
# Model pergerakan: Random Waypoint (sederhana dan umum)
Group.movementModel = RandomWaypoint
# Router: DecisionEngineRouter (memungkinkan penggunaan DecisionEngine)
Group.router = DecisionEngineRouter
# Ukuran buffer pesan (bytes) - Sesuaikan dengan kebutuhan Anda. Penting untuk Epidemic!
Group.bufferSize = 5M
# Waktu tunggu (detik) setelah mencapai tujuan (acak antara min dan max)
Group.waitTime = 0, 120
# Kecepatan pergerakan (m/s) (acak antara min dan max)
Group.speed = 0.5, 1.5
# Jumlah interface yang dimiliki setiap host (dalam kasus ini, hanya 1)
Group.nrofInterfaces = 1
# Interface yang digunakan (dalam kasus ini, Bluetooth)
Group.interface1 = btInterface
# Time-to-Live (TTL) pesan (menit) - Penting untuk membatasi penyebaran pesan!
Group.msgTtl = 300
# Jumlah host dalam grup
Group.nrofHosts = 100

# Pengaturan khusus untuk DecisionEngineRouter dan Epidemic
# Menggunakan EpidemicDecisionEngine
DecisionEngineRouter.decisionEngine = EpidemicDecisionEngine

# Pengaturan khusus untuk Grup 1 (satu-satunya grup)
# ID grup (dapat digunakan untuk membedakan grup dalam laporan)
Group1.groupID = m

## Pengaturan Pembuatan Pesan
# Jumlah event generator (sumber pesan)
Events.nrof = 1
# Class event generator yang digunakan (MessageEventGenerator membuat pesan acak)
Events1.class = MessageEventGenerator
# Interval pembuatan pesan (detik) (acak antara min dan max)
Events1.interval = 25,35
# Ukuran pesan (bytes) (acak antara min dan max)
Events1.size = 500k,1M
# Jangkauan host pengirim dan penerima pesan (acak di antara semua host)
Events1.hosts = 0,99
# Prefix ID pesan (memudahkan untuk mengidentifikasi pesan dari generator ini)
Events1.prefix = M

## Pengaturan Model Pergerakan (Random Waypoint)
# Ukuran dunia simulasi (meter)
MovementModel.worldSize = 300, 300
# Waktu pemanasan (detik) sebelum simulasi "nyata" dimulai
MovementModel.warmup = 1
